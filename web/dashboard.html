package com.novus.bc;



public class BCCalc
{
    static enum Operator
    {
        ADD("+"), SUBTRACT("-"), MULTIPLY("*"), DIVIDE("/");

        private final String op;

        private Operator(String s)
        {
            op = s;
        }

        public String getOp()
        {
            return op;
        }

        public static Operator getEnum(String value)
        {
            for (Operator v : values())
                if (v.getOp().equalsIgnoreCase(value))
                    return v;
            return null;
        }
    };

    static class Node
    {
        Operator operator;
        Node left;
        Node right;
        T value;

        public Node(T value)
        {
            super();
            this.value = value;
        }

        public Node(Operator operator, Node left, Node right)
        {
            super();
            this.operator = operator;
            this.left = left;
            this.right = right;
        }

    }

    interface Parser
    {
        Node parse(String exp);
    }

    static class SimpleParser implements Parser
    {
        @Override
        public Node parse(String exp)
        {
            return parseExpression(new StringBuffer(exp));
        }

        private Node parseExpression(StringBuffer sb)
        {
            Node num = parseMultiDivExp(sb);
            Operator operator = getOperator(sb);
            if (Operator.ADD.equals(operator) || Operator.SUBTRACT.equals(operator))
            {
                sb.delete(sb.length() - 1, sb.length());
                Node node = parseExpression(sb);
                return new Node(operator, node, num);
            }
            return num;
        }

        private Node parseMultiDivExp(StringBuffer sb)
        {
            Node num = parseNumNode(sb);
            Operator operator = getOperator(sb);
            if (Operator.MULTIPLY.equals(operator) || Operator.DIVIDE.equals(operator))
            {
                sb.delete(sb.length() - 1, sb.length());
                Node node = parseMultiDivExp(sb);
                return new Node(operator, node, num);
            }
            return num;
        }

        private static Node parseNumNode(StringBuffer sb)
        {
            for (int i = sb.length() - 1; i >= 0; i--)
            {
                if (sb.charAt(i) < '0' || sb.charAt(i) > '9')
                {
                    String num = sb.substring(i + 1, sb.length());
                    sb.delete(i + 1, sb.length());
                    return new Node(Integer.parseInt(num));
                }
            }
            return new Node(Integer.parseInt(sb.toString()));
        }

        static Operator getOperator(StringBuffer sb)
        {
            String op = String.valueOf(sb.charAt(sb.length() - 1));
            return Operator.getEnum(op);
        }
    }

    interface Evaluator
    {
        T evaluate(Node root) throws Exception;
    }

    static class BasicEvaluator implements Evaluator
    {
        @Override
        public Integer evaluate(Node node) throws Exception
        {
            if (node.operator == null)
                return node.value;
            if (node.operator.equals(Operator.MULTIPLY))
                return evaluate(node.left) * evaluate(node.right);
            else if (node.operator.equals(Operator.DIVIDE))
                return evaluate(node.left) / evaluate(node.right);
            else if (node.operator.equals(Operator.ADD))
                return evaluate(node.left) + evaluate(node.right);
            else if (node.operator.equals(Operator.SUBTRACT))
                return evaluate(node.left) - evaluate(node.right);
            throw new Exception("Error evaluting");
        }
    }

    public static void main(String[] args) throws Exception
    {
        // String exp = "5+2-1*5+3";
        String exp = args[0];
    
        Parser parser = new SimpleParser();
        Node root = parser.parse(exp);
    
        Evaluator evaluator = new BasicEvaluator();
        int result = evaluator.evaluate(root);
    
        System.out.println(result);
    }
}
